void nLetterTriangle(int n) {
    // Write your code here.
    for(int i=1; i<=n; i++)
    {
        int v=65;
        for(int j=0; j<n-i+1; j++)
        {
            cout<<char(v);
            cout<<" ";
            v=v+1;
        }
        cout<<endl;
    }
}

A B C
A B
